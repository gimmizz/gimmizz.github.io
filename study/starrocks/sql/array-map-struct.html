SQL — Array · Map · Struct

ARRAY

-- 컬럼 선언
CREATE TABLE t (
  id BIGINT,
  tags ARRAY<STRING>,
  scores ARRAY<DOUBLE>
)
PRIMARY KEY(id) DISTRIBUTED BY HASH(id);

-- 리터럴
SELECT [1,2,3], ARRAY[1,2,3];

-- 길이 / 인덱싱(1-base)
SELECT array_length(tags), tags[1];

-- 자주 쓰는 함수
SELECT array_contains(tags, 'a'),
       array_position(tags, 'a'),
       array_distinct(tags),
       array_sort(tags),
       array_concat(tags, ['x']),
       array_intersect(tags, ['a','b']),
       array_slice(tags, 1, 3);

-- 람다 기반
SELECT array_filter(scores, x -> x > 50),
       array_map(scores, x -> x * 2),
       array_sum(scores),
       array_avg(scores);

UNNEST — 배열을 행으로

SELECT t.id, tag
FROM t, UNNEST(t.tags) AS u(tag);

-- 다중 컬럼 동시 unnest
SELECT t.id, tag, score
FROM t, UNNEST(t.tags, t.scores) AS u(tag, score);

MAP

CREATE TABLE m (
  id BIGINT,
  attrs MAP<STRING, STRING>
);

-- 리터럴
SELECT map{'a':'1','b':'2'};

-- 접근
SELECT attrs['name'];
SELECT map_keys(attrs), map_values(attrs);
SELECT map_size(attrs);

-- map → 행
SELECT id, k, v
FROM m, UNNEST(map_keys(attrs), map_values(attrs)) AS u(k, v);

STRUCT

CREATE TABLE s (
  id BIGINT,
  addr STRUCT<city STRING, zip STRING>
);

-- 접근
SELECT addr.city, addr.zip FROM s;

-- 리터럴
SELECT named_struct('city','Seoul','zip','06000');
SELECT row('Seoul','06000');

LATERAL JOIN (UNNEST 포함)

SELECT t.id, x.tag, x.idx
FROM t,
     LATERAL (SELECT tag, ordinal AS idx
              FROM UNNEST(t.tags) WITH ORDINALITY u(tag, ordinal)) x;
클러스터 적용 노트
  • 이벤트 페이로드의 다중 태그/속성을 행으로 풀어보고 싶을 때 ARRAY + UNNEST가 가장 자연스럽습니다.
  • 큰 ARRAY(수백 원소+)는 압축률·쿼리 비용 모두 부담 — 가능하면 정규화하거나 핵심 키만 컬럼화.
