SQL-2. DML — INSERT / UPDATE / DELETE

INSERT

-- VALUES
INSERT INTO mart.orders VALUES (1, 100, 'paid', 12345.00, NOW());

-- 다중 행
INSERT INTO mart.orders VALUES
  (2, 100, 'paid', 5000, NOW()),
  (3, 101, 'pending', 200, NOW());

-- INSERT INTO ... SELECT (가장 흔함)
INSERT INTO mart.orders
SELECT * FROM hive_prod.dwh.orders WHERE dt = '2026-05-07';

-- 외부 카탈로그로도 INSERT (Hive v3.2+, Iceberg v3.x)
INSERT INTO hive_prod.dwh.summary SELECT ... ;

-- INSERT OVERWRITE (파티션 단위 덮어쓰기)
INSERT OVERWRITE mart.orders PARTITION (p20260507)
SELECT * FROM stage.orders WHERE updated_at::date = '2026-05-07';

-- 라벨 (재시도 안전)
INSERT INTO mart.orders WITH LABEL ord_20260507_001 SELECT ...;

UPDATE (Primary Key 테이블에 한해 직접 가능)

UPDATE mart.orders SET status = 'refunded' WHERE order_id = 1;

UPDATE mart.orders
SET amount = amount * 0.9
WHERE updated_at >= '2026-05-01' AND status = 'paid';

-- JOIN 조건으로 갱신
UPDATE mart.orders o
SET status = 'closed'
FROM stage.cancel s
WHERE o.order_id = s.order_id;

Duplicate / Aggregate / Unique 모델은 직접 UPDATE 불가. UPSERT/REPLACE 적재로 대신.

DELETE

DELETE FROM mart.orders WHERE updated_at < '2026-01-01';

-- PK 테이블이면 일반 DELETE 가능. 비-PK 테이블은 파티션 단위 권장.
ALTER TABLE mart.orders DROP PARTITION p20260101;

CTE / WITH

WITH paid AS (
  SELECT user_id, sum(amount) amt FROM mart.orders WHERE status='paid' GROUP BY user_id
),
ranked AS (
  SELECT user_id, amt, rank() OVER (ORDER BY amt DESC) rk FROM paid
)
SELECT * FROM ranked WHERE rk <= 100;

EXPLAIN / EXPLAIN ANALYZE

EXPLAIN  SELECT count(*) FROM mart.orders WHERE updated_at >= '2026-05-07';
EXPLAIN ANALYZE SELECT ...;
EXPLAIN ALL SELECT ...;     -- 모든 단계 (parser, analyzer, optimizer)

클러스터 적용 노트
  • 대용량 백필은 INSERT OVERWRITE를 파티션 단위로 끊어 진행. 한 번에 전체 OVERWRITE는 위험.
  • UPDATE/DELETE는 PK 테이블 전용. ETL 업스트림에서 변경 분만 보내고 PK 테이블이 받아주는 패턴이 안전·효율적.
  • Async MV가 있는 베이스 테이블에 큰 DML이 일어나면 MV가 무효화될 수 있음 — 의도적인 큰 변경 후엔 REFRESH MATERIALIZED VIEW를 명시적으로.
