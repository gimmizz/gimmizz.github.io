SQL-7. JSON

JSON 타입

CREATE TABLE event_log (
  event_id BIGINT NOT NULL,
  ts       DATETIME,
  payload  JSON
)
DUPLICATE KEY(event_id, ts)
DISTRIBUTED BY HASH(event_id);

INSERT INTO event_log VALUES
  (1, NOW(), parse_json('{"user_id": 100, "action":"click", "tags":["a","b"]}'));

추출

SELECT
  json_query(payload, '$.user_id')           AS user_id_json,    -- JSON
  get_json_string(payload, '$.action')       AS action_str,
  get_json_int(payload, '$.user_id')         AS user_id_int,
  get_json_double(payload, '$.amount')       AS amount,
  json_string(json_query(payload, '$.user_id'))  AS as_string
FROM event_log;

존재 / 배열

SELECT json_exists(payload, '$.tags');       -- bool
SELECT json_array_length(payload, '$.tags');
SELECT json_query(payload, '$.tags[0]');     -- 첫 원소
SELECT json_query(payload, '$.tags[*]');     -- 전체

키 목록 / 객체

SELECT json_keys(payload);                   -- ['user_id','action','tags']
SELECT json_keys(payload, '$.user');         -- 특정 경로 하위 키들
SELECT json_object('user_id', 100, 'action', 'click');
SELECT json_array(1, 2, 'three', NULL);

변경

SELECT json_set(payload, '$.action', 'view');
SELECT json_remove(payload, '$.tags');
SELECT json_insert(payload, '$.user_grade', 'gold');  -- 없을 때만
SELECT json_replace(payload, '$.action', 'view');     -- 있을 때만

파싱 / 직렬화

SELECT parse_json('{"a":1,"b":2}');        -- 문자열 → JSON
SELECT json_string(j);                     -- JSON → 문자열
SELECT cast(j AS STRING);                  -- 동일

배열을 행으로 펼치기 (UNNEST)

SELECT e.event_id, t.tag
FROM event_log e,
     unnest(cast(json_query(e.payload, '$.tags[*]') AS ARRAY<STRING>)) AS t(tag);
클러스터 적용 노트
  • 핫 키는 컬럼화: 자주 조회하는 JSON 키는 적재 단계에서 별도 컬럼으로 추출해 두는 게 빠르고 압축률도 좋음. JSON은 유연성 ↔ 비용 trade-off.
  • v3.x부터 JSON 타입의 인덱스/통계가 점차 강해지고 있음 — 사용 버전의 릴리스 노트 확인.
  • Routine Load + JSON: Kafka에서 JSON 메시지를 받을 때 jsonpaths로 미리 컬럼 매핑(L-2).
