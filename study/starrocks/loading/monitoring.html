자주 쓰는 진단 쿼리

최근 1시간 실패 적재

SELECT db_name, label, type, state, create_time, finish_time, error_msg
FROM information_schema.loads
WHERE state IN ('CANCELLED','FAILED')
  AND create_time >= NOW() - INTERVAL 1 HOUR
ORDER BY create_time DESC;

잡 종류별 처리량 / 실패율

SELECT
  type,
  count(*) AS total,
  sum(state = 'FINISHED') AS finished,
  sum(state IN ('CANCELLED','FAILED')) AS failed,
  ROUND(sum(state IN ('CANCELLED','FAILED')) / count(*), 4) AS fail_rate
FROM information_schema.loads
WHERE create_time >= NOW() - INTERVAL 24 HOUR
GROUP BY type;

적재 지연 측정 (Routine Load)

# 데이터의 timestamp 컬럼 vs 현재 시각
SELECT max(updated_at) AS last_loaded_ts,
       NOW() - max(updated_at) AS lag
FROM mart.orders;

# Kafka offset lag와 함께 보면 더 정확

Routine Load PAUSED 검출

SELECT name, state, error_log_urls, statistic
FROM information_schema.routine_loads
WHERE state = 'PAUSED';

Pipe 실패 파일

SELECT pipe_name, file_name, load_state, file_size, error_msg
FROM information_schema.pipe_files
WHERE load_state = 'ERROR'
ORDER BY last_modified DESC LIMIT 50;

Audit Log를 활용한 적재 분석

-- audit_loader 플러그인 적재 후
SELECT user, count(*) AS jobs,
       AVG(query_time)/1000 AS avg_sec,
       SUM(scan_rows) AS total_rows
FROM starrocks_audit_db__.starrocks_audit_tbl__
WHERE event_time >= NOW() - INTERVAL 1 DAY
  AND query_type IN ('LOAD','INSERT')
GROUP BY user
ORDER BY jobs DESC;
