클러스터 상태

SHOW FRONTENDS;
SHOW BACKENDS;

SHOW VARIABLES;
SHOW PROC '/statistic';                 -- 클러스터 요약

SHOW PROC '/backends/<id>/disks';        -- BE 디스크별
SHOW PROC '/colocation_group';           -- Colocate 그룹
SHOW PROC '/current_queries';

-- 노드 alive 빠른 확인
SELECT count(*) AS fe_alive FROM information_schema.frontends WHERE alive='true';
SELECT count(*) AS be_alive FROM information_schema.backends  WHERE alive='true';

노드 추가 / 제거

ALTER SYSTEM ADD FOLLOWER  "fe-new:9010";
ALTER SYSTEM ADD OBSERVER  "fe-obs:9010";
ALTER SYSTEM ADD BACKEND   "be-new:9050";

-- 안전 제거 절차
ALTER SYSTEM DECOMMISSION BACKEND "old-be:9050";
-- TabletNum이 0이 될 때까지 대기 후
ALTER SYSTEM DROP BACKEND "old-be:9050";

ALTER SYSTEM DROP FOLLOWER "fe-old:9010";

사용자/권한

SET PASSWORD FOR 'analyst' = PASSWORD('newpass');
DROP USER 'analyst';

CREATE ROLE bi_reader;
GRANT SELECT ON ALL TABLES IN DATABASE mart TO ROLE bi_reader;
GRANT USAGE ON CATALOG hive_prod TO ROLE bi_reader;
GRANT bi_reader TO USER 'analyst';

SHOW ROLES;
SHOW GRANTS FOR 'analyst';

백업/복원

SHOW REPOSITORIES;
BACKUP SNAPSHOT mart.snap_20260507 TO backup_hdfs ON (orders) PROPERTIES ("type" = "full");
SHOW BACKUP FROM mart\G

RESTORE SNAPSHOT mart_restored.snap_20260507 FROM backup_hdfs ON (orders)
PROPERTIES ("backup_timestamp" = "...", "replication_num" = "3");
SHOW RESTORE FROM mart_restored\G
